#LaunchDescription : Launch脚本行为的集合 对象
from launch import LaunchDescription 

# Node是Launch脚本的一个行为，行为可以被翻译为 ros2 run
from launch_ros.actions import Node

# FindPackageShare 是指找到install文件夹下某个包的share文件夹
# 补充： share文件夹是包的需要的文件或者配置文件的存放位置
#       lib文件时包可执行文件或者库的存放位置
from launch_ros.substitutions import FindPackageShare

# LaunchConfiguration 是指这个launch脚本需要指定一个叫config的参数
# 比如 ros2 launch rmcs_bringup xxx.launch.pt config:=a
from launch.substitutions import LaunchConfiguration, PathJoinSubstitution

# 默认执行函数
def generate_launch_description():
    ld = LaunchDescription()

    ld.add_action(
        Node(
            package="rmcs_executor",
            executable="rmcs_executor",
            parameters=[

                # FindPackageShare函数用于找到ros2 package的路径，
                # PathJoinSubstitution会将这个绝对路径和后面的字符
                # 串进行整合，从而得到完整的包内文件路径
                # 这是ros2 launch中常用的技巧
                PathJoinSubstitution(
                    [
                        FindPackageShare("rmcs_bringup"),
                        "config",
                        LaunchConfiguration("config"),
                    ]
                )
            ],

            # 异常停止后自动重启
            respawn=True,
            # prefix=["gdb -ex run --args"],
            # 自动重启的时间
            respawn_delay=5.0,
            # 这是因为ros会因为一些未知自动停下运行，或者人为错误
            # 自动重启可以防止机器人直接寄掉
        )
    )

    return ld
